ফ্রন্ট এন্ড কে ব্যাক এন্ড এর সাথে কানেক্ট করা।

সর্বপ্রথম রিয়েক্ট এপ্ল্যিকেশন কে বিল্ড করে নিতে হবে। কমান্ড npm run build , যতবার এপ্ল্যিকেশন আপডেট করবো ততবার এপ্ল্যিকেশন কে বিল্ড করে নিতে হবে।

ফ্রন্ট এন্ডের app.js ফাইলের বিতরে নিম্নের কোড গুলো লিখতে হবে।

app.use(express.static("./Client_Side/build"))

app.get("*", function(req, res){
    req.sendFile(path.resolve(__dirname, "./Client_Side", "build", "index.html"))
})

উপরের কোড কপি করে ব্যাবহার করা যাবে।

এই proxy শুধু মাত্র আমাদের কাজ করার সময় লাগবে, এপ্ল্যিকেশন সার্ভারে রান করা সময় লাগবে না। "proxy": "http://localhost:5000"
